### Step2: Train the model

    --train-file "path_to_train_file" \
    --eval-file "path_to_eval_file" \
    --outputs-dir "path_to_outputs_file" \
    --scale 3 \
    --lr 1e-4 \
    --batch-size 16 \
    --num-epochs 400 \
    --num-workers 0 \
    --seed 123
